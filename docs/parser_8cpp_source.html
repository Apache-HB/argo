109 unique_ptr<json> parser::parse_value(lexer &l, size_t nesting_depth)
110 {
111  const token &t = l.next();
112 
113  switch (t.get_type())
114  {
115  case token::begin_object_e:
116  return parse_object(l, nesting_depth + 1);
117  case token::begin_array_e:
118  return parse_array(l, nesting_depth + 1);
119  case token::number_int_e:
120  if (m_convert_numbers)
121  {
122  return parse_number_int(t);
123  }
124  else
125  {
126  return unique_ptr<json>(new json(json::number_int_e, t.get_raw_value()));
127  }
128  case token::number_double_e:
129  if (m_convert_numbers)
130  {
131  return parse_number_double(t);
132  }
133  else
134  {
135  return unique_ptr<json>(new json(json::number_double_e, t.get_raw_value()));
136  }
137  case token::string_e:
138  if (m_convert_strings)
139  {
140  try
141  {
142  return unique_ptr<json>(new json(utf8::json_string_to_utf8(t.get_raw_value())));
143  }
144  catch (json_utf8_exception &e)
145  {
146  e.add_byte_index(m_reader.get_byte_index());
147  throw e;
148  }
149  }
150  else
151  {
152  return unique_ptr<json>(new json(json::string_e, t.get_raw_value()));
153  }
154  case token::false_e:
155  return unique_ptr<json>(new json(false));
156  case token::true_e:
157  return unique_ptr<json>(new json(true));
158  case token::null_e:
159  return unique_ptr<json>(new json);
160  default:
161  throw json_parser_exception(
162  json_parser_exception::unexpected_token_e,
163  t.get_raw_value(),
164  m_reader.get_byte_index());
165  }
166 }

168 unique_ptr<json> parser::parse_array(lexer &l, size_t nesting_depth)
169 {
170  if (nesting_depth > m_max_nesting_depth)
171  {
172  throw json_parser_exception(
173  json_parser_exception::maximum_nesting_depth_exceeded_e,
174  m_max_nesting_depth,
175  m_reader.get_byte_index());
176  }
177 
178  unique_ptr<json> array = unique_ptr<json>(new json(json::array_e));
179 
180  const token &t1 = l.next();
181 
182  if (t1.get_type() == token::end_array_e)
183  {
184  return array;
185  }
186  else
187  {
188  l.put_back_last();
189  array->get_array().push_back(parse_value(l, nesting_depth));
190  }
191 
192  while (true)
193  {
194  const token &t2 = l.next();
195 
196  if (t2.get_type() == token::value_separator_e)
197  {
198  array->get_array().push_back(parse_value(l, nesting_depth));
199  }
200  else if (t2.get_type() == token::end_array_e)
201  {
202  return array;
203  }
204  else
205  {
206  throw json_parser_exception(
207  json_parser_exception::unexpected_token_e,
208  t2.get_raw_value(),
209  m_reader.get_byte_index());
210  }
211  }
212 
213  return array;
214 }

216 void parser::parse_name_value_pair(lexer &l, unique_ptr<json> &object, size_t nesting_depth)
217 {
218  const token &t1 = l.next();
219 
220  string name;
221 
222  if (t1.get_type() == token::string_e)
223  {
224  if (m_convert_strings)
225  {
226  name = *utf8::json_string_to_utf8(t1.get_raw_value());
227  }
228  else
229  {
230  name = t1.get_raw_value();
231  }
232  }
233  else
234  {
235  throw json_parser_exception(
236  json_parser_exception::unexpected_token_e,
237  t1.get_raw_value(),
238  m_reader.get_byte_index());
239  }
240 
241  const token &t2 = l.next();
242 
243  if (t2.get_type() != token::name_separator_e)
244  {
245  throw json_parser_exception(
246  json_parser_exception::unexpected_token_e,
247  t2.get_raw_value(),
248  m_reader.get_byte_index());
249  }
250 
251  object->get_object()[name] = parse_value(l, nesting_depth);
252 }

254 unique_ptr<json> parser::parse_object(lexer &l, size_t nesting_depth)
255 {
256  if (nesting_depth > m_max_nesting_depth)
257  {
258  throw json_parser_exception(
259  json_parser_exception::maximum_nesting_depth_exceeded_e,
260  m_max_nesting_depth,
261  m_reader.get_byte_index());
262  }
263 
264  unique_ptr<json> object = unique_ptr<json>(new json(json::object_e));
265 
266  const token &t1 = l.next();
267 
268  // check for empty object
269  if (t1.get_type() == token::end_object_e)
270  {
271  return object;
272  }
273  else
274  {
275  l.put_back_last();
276  }
277 
278  while (true)
279  {
280  parse_name_value_pair(l, object, nesting_depth);
281 
282  const token &t2 = l.next();
283 
284  if (t2.get_type() == token::end_object_e)
285  {
286  return object;
287  }
288  else if (t2.get_type() == token::value_separator_e)
289  {
290  continue;
291  }
292  else
293  {
294  throw json_parser_exception(
295  json_parser_exception::unexpected_token_e,
296  t2.get_raw_value(),
297  m_reader.get_byte_index());
298  }
299  }
300 
301  return object;
302 }

304 unique_ptr<json> parser::parse()
305 {
306  m_reader.reset_byte_index();
307 
308  lexer l(m_reader, m_max_token_length);
309 
310  auto res = parse_value(l, 0);
311 
312  if (m_read_all)
313  {
314  // check that there isn't anything other than whitespace left
315  int c;
316 
317  while ((c = m_reader.next()) != EOF)
318  {
319  if ((c != 0x20) && (c != 0x09) && (c != 0x0A) && (c != 0x0D))
320  {
321  throw json_parser_exception(json_parser_exception::invalid_character_e, m_reader.get_byte_index());
322  }
323  }
324  }
325  return res;
326 }
